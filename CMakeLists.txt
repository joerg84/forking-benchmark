cmake_minimum_required (VERSION 2.6)
SET(PROJECT_PREFIX "Forking-Benchmark")
PROJECT(${PROJECT_PREFIX})

message(STATUS "CMAKE_SYSTEM_NAME          = ${CMAKE_SYSTEM_NAME}"         )
message(STATUS "CMAKE_CXX_COMPILER_ID      = ${CMAKE_CXX_COMPILER_ID}"     )
message(STATUS "CMAKE_CXX_COMPILER_VERSION = ${CMAKE_CXX_COMPILER_VERSION}")
message(STATUS "CMAKE_COMPILER_IS_GNUCXX   = ${CMAKE_COMPILER_IS_GNUCXX}"  )

set (CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++11  -Wall -pedantic -Wextra -Werror ")

SET(BIN_OUTPUT_DIR ${${PROJECT_PREFIX}_BINARY_DIR}/bin)
SET(LIB_OUTPUT_DIR ${${PROJECT_PREFIX}_BINARY_DIR}/lib)

SET(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${BIN_OUTPUT_DIR})
SET(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${BIN_OUTPUT_DIR})
SET(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${LIB_OUTPUT_DIR})

FILE(COPY scripts/run_test.py DESTINATION scripts)
FILE(COPY scripts/graph.plot DESTINATION scripts)
FILE(COPY scripts/run_clone_test.py DESTINATION scripts)
FILE(COPY scripts/clone_graph.plot DESTINATION scripts)

ENABLE_TESTING()
ADD_SUBDIRECTORY(src)
